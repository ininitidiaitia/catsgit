            if ( recursive ) {
                // getDirectoryMembersメソッドをrecursive引数trueで呼び出し、VCDirectoryMemberのListを取得
                final List<VCDirectoryMember> directoryMemberList = getDirectoryMembers( vCAuthentication, accessorParameter, fromRevision, toRevision, true );
                // 取得したVCDirectoryMemberのListで処理をループし、情報をGitResourceInfoに格納する
                for ( final VCDirectoryMember directoryMember : directoryMemberList ) {
                    final GitResourceInfo resourceInfo = new GitResourceInfo();
                    final GitRevision finalRevision = new GitRevision( RevisionType.OBJECT_ID, directoryMember.getLastModifiedRevision() );
                    String encodeUri = makeReturnRepoPathStr( accessorParameter.getProviderURI(), accessorParameter );
                    encodeUri = addRelativePath( encodeUri, accessorParameter );
                    final GitURI gituri = new GitURI( accessorParameter.getProviderURI(), URI.create( accessorParameter.getProviderURI() ) );
                    resourceInfo.setFileNodeType( directoryMember.getFileNodeType() );
                    resourceInfo.setLastCommitAuthor( directoryMember.getLastCommitAuthor() );
                    resourceInfo.setLastModifiedDate( directoryMember.getLastModifiedDate() );
                    resourceInfo.setLastModifiedRevision( finalRevision );
                    resourceInfo.setRevision( finalRevision );
                    resourceInfo.setPath( encodeUri );
                    resourceInfo.setVCURI( new VCURI( URI.create( encodeUri ) ) );
                    resourceInfo.setRepository( gituri );

                    resourceInfoList.add( resourceInfo );
                }
            }
